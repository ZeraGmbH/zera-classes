{

  "value_sequence": [
    <Name of 1st value in "data" arrays>,
    <Name of 2nd value in "data" arrays>,
    ...
  ],
  "values": [
    {
      "data": [
        <1st value>,
        <2nd value>,
        ...
      ],
      "ts": <ms since first value>
    },
    {
      "data": [
        <1st value>,
        <2nd value>,
        ...
      ],
      "ts": <ms since first value>
    },
    ...
  ]
